üéØ DAY 1: Two Pointers

"The Squeeze Play"

üé¨ THE STORY: The Sandwich Maker

Imagine you're making the PERFECT sandwich. You have two hands:

  • LEFT HAND starts at the bottom slice of bread
  • RIGHT HAND starts at the top slice of bread

Your job? Squeeze them together to find the perfect ingredients in the middle!

Sometimes both hands move toward each other (like finding if bread slices match). Sometimes one hand stays still while the other explores (like spreading peanut butter from left to right).

This is Two Pointers - using two references that move through your data to solve problems efficiently!

üíª CODE TEMPLATES

Template 1: The Squeeze (Opposite Ends)

function twoPointersSqueeze(arr) {
    let left = 0;
    let right = arr.length - 1;
    
    while (left < right) {
        // Check your condition
        if (/* condition met */) {
            // Found it!
            return true;
        } else if (/* need smaller value */) {
            right--;
        } else {
            left++;
        }
    }
    
    return false;
}

Template 2: The Chase (Same Direction)

function twoPointersChase(arr) {
    let slow = 0; // Builds result
    
    for (let fast = 0; fast < arr.length; fast++) {
        if (/* condition met */) {
            arr[slow] = arr[fast];
            slow++;
        }
    }
    
    return slow;
}

function twoSum(numbers, target) {
    let left = 0;
    let right = numbers.length - 1;
    
    while (left < right) {
        const sum = numbers[left] + numbers[right];
        
        if (sum === target) {
            return [left, right];
        } else if (sum < target) {
            left++; // Need bigger sum
        } else {
            right--; // Need smaller sum
        }
    }
    
    return [-1, -1];
}

// Time: O(n), Space: O(1)

Why it works: In a sorted array, if sum is too small, we MUST move left pointer right. If sum is too big, we MUST move right pointer left!

function removeDuplicates(nums) {
    if (nums.length === 0) return 0;
    
    let slow = 0;
    
    for (let fast = 1; fast < nums.length; fast++) {
        if (nums[fast] !== nums[slow]) {
            slow++;
            nums[slow] = nums[fast];
        }
    }
    
    return slow + 1;
}

// Time: O(n), Space: O(1)

function isPalindrome(s) {
    const isAlphaNum = (char) => {
        const code = char.charCodeAt(0);
        return (code >= 48 && code <= 57) ||
               (code >= 65 && code <= 90) ||
               (code >= 97 && code <= 122);
    };
    
    let left = 0;
    let right = s.length - 1;
    
    while (left < right) {
        while (left < right && !isAlphaNum(s[left])) {
            left++;
        }
        while (left < right && !isAlphaNum(s[right])) {
            right--;
        }
        
        if (s[left].toLowerCase() !== s[right].toLowerCase()) {
            return false;
        }
        
        left++;
        right--;
    }
    
    return true;
}

// Time: O(n), Space: O(1)
